    QWriteLocker loker(&s_lock);
    bool result = false;
    if(s_quick_event_pool.contains(eventName)) {
        QMap<qint32, QObject *> tmplist = s_quick_event_pool[eventName];
        tmplist.remove(tmplist.key(listener));
        s_quick_event_pool.insert(eventName, tmplist);
        result = true;

        QMap<qint32, QObject *> tmplist = s_quick_event_pool[eventName];

        s_quick_event_pool.insert(eventName, tmplist);
